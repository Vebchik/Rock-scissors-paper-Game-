import random #Импортируем библиотеку рандома
list_of_elements = ["Камень", "Ножницы", "Бумага"] #Создаём список элементов для программы
random_element = random.choice(list_of_elements) #Программа будет выбирать случайный элемент из списка
def rock_script(random_element): #Функция на случай, если игрок выбрал камень
    if random_element == "Камень":
        print("Камень!\nУпс, кажется, ничья.")
    elif random_element == "Ножницы":
        print("Ножницы!\nПоздравляю, ты победил!!!")
    elif random_element == "Бумага":
        print("Бумага!\nК сожалению, ты проиграл.")
def scissors_script(random_element): #Функция на случай, если игрок выбрал ножницы
    if random_element == "Камень":
        print("Камень!\nК сожалению, ты проиграл.")
    elif random_element == "Ножницы":
        print("Ножницы!\nУпс, кажется, ничья.")
    elif random_element == "Бумага":
        print("Бумага!\nПоздравляю, ты победил!!!")
def paper_script(random_element): #Функция на случай, если игрок выбрал бумагу
    if random_element == "Камень":
        print("Камень!\nПоздравляю, ты победил!!!")
    elif random_element == "Ножницы":
        print("Ножницы!\nК сожалению, ты проиграл.")
    elif random_element == "Бумага":
        print("Бумага!\nУпс, кажется, ничья.")
if input("Привет! Хочешь поиграть? ") == "Да": #Далее запускается скрипт диалога с игроком
    if input("У меня есть одна игра для тебя, она называется \"Камень, ножницы, бумага\", хочешь сыграть? ") == "Да":
        if input("Правила просты. Есть три предмета: камень, ножницы, бумага.\nНасчёт \"Три\" мы одновременно выбираем один из трёх предметов.\nПобеждает тот, чей предмет бьёт предмет противника.\nКамень побеждает ножницы, ножницы побеждают бумагу, а бумага побеждает камень.\nНу что, готов? ") == "Да":
            users_choise = input("Камень, ножницы, бумага. 1, 2, 3 ") #Вот здесь начинается основной игровой скрипт
            if users_choise == "Камень":
                rock_script(random_element)
            elif users_choise == "Ножницы":
                scissors_script(random_element)
            elif users_choise == "Бумага":
                paper_script(random_element)
            else:
                print("Такого предмета нет в правилах игры, ты проиграл")
        else:
            print("Хорошо. Напиши мне, когда будешь готов :)")
    else:
        print("Хорошо. Напиши мне, когда захочешь :)")
else:
    print("Ладно, тогда пока :)")
